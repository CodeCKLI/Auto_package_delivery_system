13   def deliver(truck):
14       first_load = [1, 6, 13, 14, 15, 16, 19, 20, 21, 22, 29, 30, 31, 34, 37, 40]
15       second_load = [2, 3, 4, 5, 9, 10, 11, 12, 17, 18, 23, 24, 25, 26, 36, 38]
16       third_load = [7, 8, 27, 28, 32, 33, 35, 39]
17   
18       if truck.truck_id == 'truck1':
19           truck.loadPackages(first_load)
20       if truck.truck_id == 'truck2':
21           truck.loadPackages(second_load)
22       if truck.truck_id == 'truck3':
23           truck.loadPackages(third_load)
24   
25       PrintColor.prGreen("===============================================================")
26       print(truck.truck_id, 'starts delivery at:', truck.start_time.time())
27       PrintColor.prGreen("===============================================================")
28   
29       while truck.packages:
30           truck.deliver_next_location()
31       truck.back_to_hub()
32   
33       PrintColor.prGreen("===============================================================")
34       print(truck.truck_id, 'Delivery route:', truck.delivery_route)
35       print(truck.truck_id, 'Miles traveled:', round(truck.miles, 2))
36       print(truck.truck_id, 'Time arrived at the Hub(in H:M format):', truck.current_time.hour, ':',
37             truck.current_time.minute)
38       PrintColor.prGreen("===============================================================")
